class Step6Service:

二、Step6 当前要升级的点
现在 Step6 已经能提取新增信息，但还不够“投资判断系统化”。需要升级为：

1. 不只提取信息，还要判断信息类型

info_type: str

fact        明确事实
claim       公司/创始人口径
number      数字/财务/订单/客户数据
plan        未来计划

novelty_type: str

new             完全新增

confidence: str
可选值：

high / medium / low
判断标准：

high：有明确数据、明确表述、原文证据强

CONFIDENCE_MAP = {
    "high": ConfidenceLevel.HIGH,
    "medium": ConfidenceLevel.MEDIUM,
    "low": ConfidenceLevel.LOW,
}

VALID_INFO_TYPES = {
    "fact", "claim", "number", "plan", "risk_signal", "correction"
}

VALID_NOVELTY_TYPES = {
    "new", "more_specific", "contradiction", "confirmation"
}


def _parse_confidence(s: str) -> ConfidenceLevel:
    if not s:
        return ConfidenceLevel.MEDIUM
    return CONFIDENCE_MAP.get(str(s).lower(), ConfidenceLevel.MEDIUM)


def _parse_info_type(s: str) -> str:
    if not s:
        return "fact"
    s = str(s).lower()
    return s if s in VALID_INFO_TYPES else "fact"


def _parse_novelty_type(s: str) -> str:
    if not s:
        return "new"
    s = str(s).lower()
    return s if s in VALID_NOVELTY_TYPES else "new"


def _parse_new_information(raw: Dict[str, Any], idx: int) -> NewInformation:
    cat_str = raw.get("category", "战略")
    cat = CATEGORY_MAP.get(cat_str, InfoCategory.STRATEGY)

    imp_str = raw.get("importance", "medium")
    imp = IMPORTANCE_MAP.get(str(imp_str).lower(), ImportanceLevel.MEDIUM)

    return NewInformation(
        id=raw.get("id", f"ni_{idx + 1:03d}"),

- more_specific：比会前判断更具体
- contradiction：与BP或会前判断矛盾
- confirmation：强化了会前判断

confidence 只能选：
- high：表述具体，有明确数字/客户/时间/证据
- medium：说法较清楚，但仍需验证
- low：模糊、宣传性强、缺乏验证

contradicts_bp：
- true：与BP或会前判断明显不一致
- false：没有明显不一致

is_critical：
- true：会明显影响后续判断
- false：普通补充信息

affects_judgement：
填写这条信息影响的判断方向，例如：
- 商业化验证
- 收入质量
- 客户真实性
- 技术壁垒
- AI平台真实性
- 团队可信度
- 产能兑现
- 扩张可行性
- 估值合理性
- 整合风险
- 竞争壁垒

related_prior_judgement：
填写它对应的会前判断。
如果没有对应，写“未匹配到明确会前判断”。

follow_up_hint：
填写后续如何验证这条信息。

def _validate_new_information(items: List[NewInformation]) -> List[NewInformation]:
    """
    Step6 输出质检：
    1. content 不能为空
    2. evidence 不能为空
    3. 去重
    4. 没有 related_prior_judgement 时补默认值
    5. 没有 follow_up_hint 时补默认值
    """

    valid = []
    seen = set()

    for item in items:
        if not item.content:
            continue

        if not item.evidence:
            continue

        dedup_key = item.content.strip()
        if dedup_key in seen:
            continue
        seen.add(dedup_key)

        if not item.related_prior_judgement:
            item.related_prior_judgement = "未匹配到明确会前判断"

        if not item.follow_up_hint:
            item.follow_up_hint = "后续需进一步核实该信息的真实性和影响。"

        valid.append(item)

    return valid

# step7_matcher.py

class Step7Matcher:

    def __init__(self, llm):
        self.llm = llm

    def run(self, questions, new_information):
        prompt = STEP7A_PROMPT.format(
            questions=self._format_questions(questions),
            new_information=self._format_info(new_information)
        )

        response = self.llm.generate(prompt)
        return self._parse(response)

    def _format_questions(self, questions):
        return "\n".join([f"q_{i+1}: {q}" for i, q in enumerate(questions)])

    def _format_info(self, infos):
        lines = []
        for ni in infos:
            lines.append(
                f"{ni['id']}: {ni['content']} (type={ni['info_type']})"
            )
        return "\n".join(lines)

    def _parse(self, response):
        import json
        return json.loads(response)

🔧 Step7 Service（整合）
# step7_service.py

class Step7Service:

    def __init__(self, llm):
        self.matcher = Step7Matcher(llm)
        self.evaluator = Step7Evaluator(llm)

    def run(self, questions, step6_output):

        new_information = step6_output["new_information"]

        # Step7A：匹配
        matches = self.matcher.run(questions, new_information)

        # Step7B：评估
        evaluation = self.evaluator.run(
            questions,
            matches,
            new_information
        )

        # 会议质量评估（简单规则）
        meeting_quality = self._compute_meeting_quality(evaluation)

        return {
            "meeting_quality": meeting_quality,
            "question_validation": evaluation["question_validation"]
        }

    def _compute_meeting_quality(self, evaluation):
        qv = evaluation["question_validation"]

        answered = sum(1 for q in qv if q["status"] == "answered")
        total = len(qv)

        ratio = answered / total if total else 0

        if ratio > 0.7:
            directness = "high"
        elif ratio > 0.4:
            directness = "medium"
        else:
            directness = "low"

        return {
            "answer_directness": directness,
            "overall_confidence": directness,
            "evasion_level": "medium",
            "evidence_strength": "medium",
            "evasion_signals": []
        }

🧠 六、Step8 Service（核心代码）
class Step8Service:

    def __init__(self, llm):
        self.llm = llm

    def run(self, hypotheses, step7_output):

        prompt = STEP8_PROMPT.format(
            hypotheses=json.dumps(hypotheses, ensure_ascii=False, indent=2),
            question_validation=json.dumps(step7_output["question_validation"], ensure_ascii=False, indent=2)
        )

        response = self.llm.generate(prompt)

        return self._parse(response)

    def _parse(self, response):
        import json
        try:
            return json.loads(response)
        except:
            return {
                "hypothesis_updates": [],
                "new_risks": [],
                "unchanged_hypotheses": []
            }

    def _rule_based_decision(self, step8_output):

        updates = step8_output.get("hypothesis_updates", [])

        weakened = 0
        slightly_weakened = 0
        uncertain = 0
        slightly_reinforced = 0

        for u in updates:
            ct = u.get("change_type", "")

            if ct == "weakened":
                weakened += 1
            elif ct == "slightly_weakened":
                slightly_weakened += 1
            elif ct == "uncertain":
                uncertain += 1
            elif ct == "slightly_reinforced":
                slightly_reinforced += 1

        # 决策逻辑
        if weakened >= 2:
            decision = "pass"
        elif weakened >= 1 and uncertain >= 2:
            decision = "hold"
        elif uncertain >= 3:
            decision = "cautious_go"
        elif slightly_reinforced >= 2:
            decision = "go"
        else:
            decision = "cautious_go"

        return {
            "decision": decision,
            "confidence": "medium",
            "one_line_conclusion": "基于当前信息的综合判断"
        }

def decide(
    step6: Dict[str, Any],
    step7: Dict[str, Any],
    step8: Dict[str, Any],
    model: str = None
) -> Dict[str, Any]:
    system_prompt = STEP9_SYSTEM
    user_prompt = STEP9_USER.format(
        step6=json.dumps(step6, ensure_ascii=False, indent=2),
        step7=json.dumps(step7, ensure_ascii=False, indent=2),
        step8=json.dumps(step8, ensure_ascii=False, indent=2)
    )

    raw = call_deepseek(system_prompt, user_prompt, model=model)
    data = _parse_json_response(raw)

    if not isinstance(data, dict):
        data = {}

    data = _apply_rule_based_guardrails(data, step6, step7, step8)
    return data


def _parse_json_response(text: str) -> Any:
    text = text.strip()
    text = re.sub(r"^```json\\s*", "", text, flags=re.IGNORECASE)
    text = re.sub(r"^```\\s*", "", text)
    text = re.sub(r"\\s*```$", "", text)
    text = text.strip()

    try:
        return json.loads(text)
    except json.JSONDecodeError:
        match = re.search(r"\\{[\\s\\S]*\\}|\\[[\\s\\S]*\\]", text)
        if match:
            return json.loads(match.group(0))
        raise


def _apply_rule_based_guardrails(
    data: Dict[str, Any],
    step6: Dict[str, Any],
    step7: Dict[str, Any],
    step8: Dict[str, Any]
) -> Dict[str, Any]:
    updates = step8.get("hypothesis_updates", [])
    validations = step7.get("question_validation", [])
    infos = step6.get("new_information", [])

    weakened = sum(1 for u in updates if u.get("change_type") in ("weakened", "slightly_weakened"))
    uncertain = sum(1 for u in updates if u.get("change_type") == "uncertain")
    reframed = sum(1 for u in updates if u.get("change_type") == "reframed")
    partially = sum(1 for v in validations if v.get("status") == "partially_answered")
    answered = sum(1 for v in validations if v.get("status") == "answered")

    verified_positive_candidates = []
    for ni in infos:

Step9 决策行动
但现在出现一个新的架构问题：

系统需要学习用户/基金偏好，但不能让个人偏好污染“项目本身判断”。

也就是说，必须区分：

项目客观判断：这个项目本身怎么样
投资人匹配判断：这个项目适不适合我 / 这只基金
比如：

政府基金要求落地、产线、反投
这不是所有投资人都适用的通用规则，不能写进通用判断库。
它应该进入“基金画像库”。

二、核心设计原则
原则 1：主判断库和偏好库必须分开
不能把：

政府基金不喜欢纯定制项目
写进通用行业判断。

否则以后系统看软件项目、AI项目、SaaS项目，可能会错误地认为：

没有产线 = 不好
这是污染。

原则 2：偏好可以影响“问什么”，但不能污染“事实是什么”
正确方式：

Step6：不调用偏好，只提取事实
Step7：可以轻调用偏好，用来检查偏好相关问题是否被回答
Step8：主要更新项目本身判断
Step10：强调用偏好，判断是否适合当前基金/投资人
一句话：

偏好影响提问和最终匹配，不影响事实提取。

原则 3：所有沉淀都先进入候选库，不自动入正式库
不能让系统自动把一次项目判断变成永久规则。

所有沉淀流程必须是：

候选沉淀 → 人工确认 → 正式入库

# services/v2/step10_fit_decider.py

import json
import re
from typing import Dict, Any
from services.deepseek_service import call_deepseek


def decide_fit(
    fund_profile: Dict[str, Any],
    step9_output: Dict[str, Any],
    project_summary: Dict[str, Any],
    user_feedback: str = "",
    model: str = None
) -> Dict[str, Any]:
    system_prompt = STEP10_SYSTEM
    user_prompt = STEP10_USER.format(
        fund_profile=json.dumps(fund_profile, ensure_ascii=False, indent=2),
        step9_output=json.dumps(step9_output, ensure_ascii=False, indent=2),
        project_summary=json.dumps(project_summary, ensure_ascii=False, indent=2),
        user_feedback=user_feedback
    )

    raw = call_deepseek(system_prompt, user_prompt, model=model)
    data = _parse_json_response(raw)

    if not isinstance(data, dict):
        data = {}

    data = _apply_fit_guardrails(data, fund_profile, step9_output)
    return data


def _parse_json_response(text: str) -> Any:
    text = text.strip()
    text = re.sub(r"^```json\\s*", "", text, flags=re.IGNORECASE)
    text = re.sub(r"^```\\s*", "", text)
    text = re.sub(r"\\s*```$", "", text)
    text = text.strip()

    try:
        return json.loads(text)
    except json.JSONDecodeError:
        match = re.search(r"\\{[\\s\\S]*\\}|\\[[\\s\\S]*\\]", text)
        if match:
            return json.loads(match.group(0))
        raise


def _apply_fit_guardrails(
    data: Dict[str, Any],
    fund_profile: Dict[str, Any],
    step9_output: Dict[str, Any]
) -> Dict[str, Any]:
    mismatches = data.get("mismatched_constraints", [])
    matched = data.get("matched_constraints", [])

    high_mismatch_count = sum(
        1 for m in mismatches if m.get("severity") == "high"
    )

    hard_constraints = fund_profile.get("hard_constraints", [])
    hard_count = len(hard_constraints)

    # 如果存在高严重度硬约束不匹配，不能给 fit
    if high_mismatch_count >= 1:
        data["fit_decision"] = "not_fit"
        data["final_recommendation"] = "pass"

    # 如果多个硬约束未匹配，直接 not_fit
    if hard_count >= 2 and len(mismatches) >= 2:
        data["fit_decision"] = "not_fit"
        data["final_recommendation"] = "pass"

    # 如果没有明显不匹配，但项目本身 not_ready，则最多 partial_fit
    investment_decision = (
        step9_output
        .get("overall_decision", {})
        .get("investment_decision", "")
    )

    if investment_decision == "not_ready" and data.get("fit_decision") == "fit":
        data["fit_decision"] = "partial_fit"
        data["final_recommendation"] = "request_materials"

    data.setdefault("fit_score", _score_fit(data))
    data.setdefault("matched_constraints", matched)
    data.setdefault("mismatched_constraints", mismatches)
    data.setdefault("compromises", [])
    data.setdefault("candidate_profile_updates", [])
    data.setdefault("candidate_case_record", {})

    return data


def _score_fit(data: Dict[str, Any]) -> int:
    decision = data.get("fit_decision")
    if decision == "fit":
        return 80
    if decision == "partial_fit":
        return 50
    if decision == "not_fit":
        return 20
    return 50

# services/knowledge/candidate_writer.py

import json
import os
from datetime import datetime
from typing import Dict, Any


class CandidateWriter:
    def __init__(self, kb_root: str):
        self.kb_root = kb_root

    def append_candidate(self, bucket: str, item: Dict[str, Any]):
        path = os.path.join(self.kb_root, "candidates", f"{bucket}.json")
        os.makedirs(os.path.dirname(path), exist_ok=True)

        data = []
        if os.path.exists(path):
            with open(path, "r", encoding="utf-8") as f:
                data = json.load(f)

        item.setdefault("status", "pending_review")
        item.setdefault("created_at", datetime.now().isoformat())

        data.append(item)

        with open(path, "w", encoding="utf-8") as f:
            json.dump(data, f, ensure_ascii=False, indent=2)

# -*- coding: utf-8 -*-
"""
profile_loader.py — 投资人 / 基金画像加载器

注意：
1. 默认只能是 neutral_investor
2. 不能默认 government_fund
3. profile 用于 Step0 / Step3 / Step4 / Step5 / Step7 / Step10
4. Step6 禁止使用 profile
"""

import json
import os
from typing import Dict, Any, List


DEFAULT_PROFILE_ID = "neutral_investor"


class ProfileLoader:
    def __init__(self, kb_root: str):
        self.kb_root = kb_root
        self.profile_dir = os.path.join(kb_root, "profiles", "fund_profiles")

    def list_profiles(self) -> List[Dict[str, Any]]:
        """列出可选 fund profiles，用于前端选择"""
        profiles = []

        if not os.path.exists(self.profile_dir):
            return profiles

        for filename in os.listdir(self.profile_dir):
            if not filename.endswith(".json"):
                continue

            path = os.path.join(self.profile_dir, filename)
            try:
                with open(path, "r", encoding="utf-8") as f:
                    data = json.load(f)

                profiles.append({
                    "profile_id": data.get("profile_id"),
                    "name": data.get("name"),
                    "description": data.get("description"),
                    "profile_type": data.get("profile_type", "fund")
                })
            except Exception:
                continue

        return profiles

    def load_profile(self, profile_id: str = None) -> Dict[str, Any]:
        """加载 profile。profile_id 为空时使用 neutral_investor"""
        if not profile_id:
            profile_id = DEFAULT_PROFILE_ID

        safe_profile_id = os.path.basename(profile_id)
        path = os.path.join(self.profile_dir, f"{safe_profile_id}.json")

        if not os.path.exists(path):
            # fallback 到 neutral
            path = os.path.join(self.profile_dir, f"{DEFAULT_PROFILE_ID}.json")

        with open(path, "r", encoding="utf-8") as f:
            return json.load(f)

            "question_id": f"q_base_{i+1:03d}",

## 问题 2：截断会影响输出质量吗？

**不会影响 Step9 的核心决策**。因为 Step9 是 **规则引擎优先** 的设计：

- why_changed（100+ 字的解释）
- updated_view（完整的新判断）

def classify_update_for_decision(update: dict) -> str:
    """
    将 Step8 的 hypothesis_direction + change_type
    转换成 Step9 可用的决策信号。

    返回：
    - validated_positive
    - confirmed_negative
    - key_uncertainty
    - neutral
    """

    direction = (update.get("hypothesis_direction") or "neutral").lower()
    change = (update.get("change_type") or "").lower()

    if change in {"uncertain", "reframed", "no_change", "unchanged"}:
        return "key_uncertainty"

    if direction == "positive":
        if change in {"reinforced", "slightly_reinforced"}:
            return "validated_positive"
        if change in {"weakened", "slightly_weakened", "overturned"}:
            return "confirmed_negative"

    if direction == "negative":
        if change in {"reinforced", "slightly_reinforced"}:
            return "confirmed_negative"
        if change in {"weakened", "slightly_weakened", "overturned"}:
            return "validated_positive"

    if direction == "neutral":
        if change in {"overturned", "weakened", "slightly_weakened"}:
            return "confirmed_negative"
        if change in {"reinforced", "slightly_reinforced"}:
            return "key_uncertainty"

    return "key_uncertainty"
